구글 스프레드시트 연동

구글 시트를 클라우드 데이터베이스로 사용합니다. Apps Script 웹 앱을 배포하고 부여받은 URL을 아래 입력해 주세요. 연동 전에는 웹 브라우저 로컬 저장소(LocalStorage)를 사용하여 단독 구동됩니다.

수동 데이터 동기화 관리

⚠️ 주의: 동기화(가져오기/내보내기) 실행 시 대상 스토리지(로컬 브라우저 또는 구글 시트)의 기존 데이터가 완전히 덮어쓰기됩니다. 실행 전 신중히 확인해 주세요.

처음이신가요? 3분 연동 방법

  1. 새 구글 스프레드시트를 만듭니다.
  2. 시트 상단 메뉴에서 [확장 프로그램] → [Apps Script]를 선택합니다.
  3. 원래 있던 코드를 지우고 아래의 코드를 복사하여 붙여넣습니다.
    /** * 농산물 매입/매출 관리 시스템 - Apps Script */ function createJsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); checkAndInitSheets(ss); var pSheet = ss.getSheetByName("매입"); var sSheet = ss.getSheetByName("매출"); return createJsonResponse({ success: true, purchases: readSheetData(pSheet), sales: readSheetData(sSheet) }); } catch (err) { return createJsonResponse({ success: false, error: err.toString() }); } } function doPost(e) { try { var postData = JSON.parse(e.postData.type === "application/json" ? e.postData.contents : e.postData.getDataAsString()); var ss = SpreadsheetApp.getActiveSpreadsheet(); checkAndInitSheets(ss); if (postData.action === "sync") { writeSheetData(ss.getSheetByName("매입"), postData.purchases); writeSheetData(ss.getSheetByName("매출"), postData.sales); return createJsonResponse({ success: true, message: "구글 시트 동기화가 성공적으로 완료되었습니다." }); } return createJsonResponse({ success: false, error: "Invalid Action" }); } catch (err) { return createJsonResponse({ success: false, error: err.toString() }); } } function checkAndInitSheets(ss) { var pSheet = ss.getSheetByName("매입"); if (!pSheet) { pSheet = ss.insertSheet("매입"); pSheet.appendRow(["id", "날짜", "품목명", "매입처", "수량", "단가", "총금액", "결제상태", "비고"]); pSheet.getRange("A1:I1").setFontWeight("bold").setBackground("#e6f4ea").setHorizontalAlignment("center"); } var sSheet = ss.getSheetByName("매출"); if (!sSheet) { sSheet = ss.insertSheet("매출"); sSheet.appendRow(["id", "날짜", "품목명", "매출처", "수량", "단가", "총금액", "수금상태", "비고"]); sSheet.getRange("A1:I1").setFontWeight("bold").setBackground("#fce8e6").setHorizontalAlignment("center"); } } function readSheetData(sheet) { var data = []; var rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return data; var headers = rows[0]; var keyMapping = { "id": "id", "날짜": "date", "품목명": "product", "매입처": "supplier", "매출처": "customer", "수량": "quantity", "단가": "unitPrice", "총금액": "totalPrice", "결제상태": "status", "수금상태": "status", "비고": "notes" }; for (var i = 1; i < rows.length; i++) { var row = rows[i]; var obj = {}; var hasData = false; for (var j = 0; j < headers.length; j++) { var headerVal = headers[j]; var key = keyMapping[headerVal] || headerVal; var cellVal = row[j]; if (cellVal instanceof Date) { cellVal = Utilities.formatDate(cellVal, Session.getScriptTimeZone(), "yyyy-MM-dd"); } obj[key] = cellVal; if (cellVal !== "") hasData = true; } if (hasData) { if (obj.quantity) obj.quantity = Number(obj.quantity); if (obj.unitPrice) obj.unitPrice = Number(obj.unitPrice); if (obj.totalPrice) obj.totalPrice = Number(obj.totalPrice); data.push(obj); } } return data; } function writeSheetData(sheet, dataList) { var lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.deleteRows(2, lastRow - 1); } if (!dataList || dataList.length === 0) return; var rows = []; for (var i = 0; i < dataList.length; i++) { var item = dataList[i]; var row = []; row.push(item.id || ""); row.push(item.date || ""); row.push(item.product || ""); row.push(item.supplier || item.customer || ""); row.push(item.quantity || 0); row.push(item.unitPrice || 0); row.push(item.totalPrice || (item.quantity * item.unitPrice) || 0); row.push(item.status || ""); row.push(item.notes || ""); rows.push(row); } sheet.getRange(2, 1, rows.length, rows[0].length).setValues(rows); }
  4. 프로젝트 저장 후, 우측 상단 [배포] → [새 배포]를 클릭합니다.
  5. 배포 유형은 웹 앱으로 지정하고, 다음과 같이 설정합니다.
    • 웹 앱을 실행할 사용자: 나 (본인 이메일)
    • 액세스 권한이 있는 사용자: 모든 사용자
  6. 배포 완료 시 표시되는 웹 앱 URL을 복사하여 왼쪽 설정창에 저장합니다.
